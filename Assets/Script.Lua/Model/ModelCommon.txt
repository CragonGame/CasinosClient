-- 银行操作通知
CardData = {}

function CardData:new(o)
    o = o or {}
    setmetatable(o, self)
    self.__index = self
    o.suit = 0
    o.type = 0
    return o
end

function CardData:setData(data)
    self.suit = data[1]
    self.type = data[2]
end

function CardData:getData4Pack()
    local p_d = {}
    table.insert(p_d, self.suit)
    table.insert(p_d, self.type)
    return p_d
end
